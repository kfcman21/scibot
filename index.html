hamster_line.py
from robomation import *

hamster = HamsterS()

while True:
    left  = hamster.left_floor()
    right = hamster.right_floor()
    # 검은 선을 따라 주행
    if left < 50 and right < 50:
        hamster.wheels(30, 30)
    elif left >= 50:
        hamster.wheels(30, 0)
    else:
        hamster.wheels(0, 30)
